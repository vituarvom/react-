## UseLocalStorage

The `UseLocalStorage` component demonstrates the use of the `useLocalStorage` hook, which simplifies working with local storage in React. This example allows users to toggle between a "dark" and "light" theme, with the selected theme saved in local storage.

## Installation

Make sure to have the React Utils Library installed:

```mdx-code-block
  <RSUTabs />
```

<ExampleCodeBlocks 
  tsCode={
    `
   import React from "react";
import { useLocalStorage } from "react-smart-utils";

    const UseLocalStorage: React.FC = () => {
  const [value, setValue] = useLocalStorage<"dark" | "light">("theme", undefined);

  console.log(value);
  return (
    <div>
      <button onClick={() => value === "light" || !value ? setValue("dark") : setValue("light")}>
        Set {value === "light" || !value ? "Dark" : "Light"} Theme
      </button>
      <p>Check the local storage to see the result</p>
    </div>
  );
};

export default UseLocalStorage;

    `

}
