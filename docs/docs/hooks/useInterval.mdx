## useInterval

The `useInterval` hook is a custom React hook that provides a declarative way to use `setInterval`. Itâ€™s helpful for managing repeated actions at a specified interval, such as polling APIs, animations, timers, and periodic state updates in functional components.

## Installation

Make sure to have the React Utils Library installed:

<ExampleCodeBlocks 
  tsCode={
    `
    import React, { useState } from 'react';
import { useInterval } from "react-smart-utils";

const UseInterval: React.FC = () => {
  const [count, setCount] = useState(0);
  const { start, clear } = useInterval(() => setCount((c) => c + 1), 1000);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={start}>Start Counting</button>
      <button onClick={clear}>Stop Counting</button>
    </div>
  );
};

export default UseInterval;

    `
